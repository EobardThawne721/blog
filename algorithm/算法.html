public int[] sortedSquares(int[] nums) {
    int result[] = new int[nums.length];
    //é¦–å°¾æŒ‡é’ˆ
    int startIndex = 0;
    int endIndex = nums.length - 1;
    int size = nums.length - 1;

    while (size >= 0) {
        //å› ä¸ºnumsæ˜¯æœ‰åºæ•°ç»„,æ‰€ä»¥å·¦å³ä¸¤è¾¹çš„å¹³æ–¹ä¸€å®šæ¯”ä¸­é—´å¤§,ä»ä¸¤è¾¹å¾€ä¸­é—´éå†ï¼ŒæŠŠæœ€å¤§çš„æ”¾åœ¨æ–°æ•°ç»„å³è¾¹
        //åˆ¤æ–­é¦–å°¾è°çš„ç»å¯¹å€¼å¤§å°±æŠŠå®ƒæ”¾åœ¨æ–°æ•°ç»„æœ«å°¾
        if (Math.abs(nums[startIndex]) > Math.abs(nums[endIndex])) {
            result[size--] = nums[startIndex] * nums[startIndex];
            startIndex++;
        } else {
            result[size--] = nums[endIndex] * nums[endIndex];
            endIndex--;
        }
    }
    return result;
}

public boolean canPartition(int[] nums) {
        int sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
        }
        //å¦‚æœæ˜¯å¥‡æ•°,è¯´æ˜ä¸èƒ½å¹³åˆ†
        if (sum % 2 != 0)
            return false;

        //numsæœ¬èº«ç±»ä¼¼äºä»·å€¼å’Œé‡é‡
        //nums:[1,2,5]

        int goodsSize = nums.length;
        //ç±»ä¼¼äºèƒŒåŒ…é‡é‡
        int weight = sum / 2;

        int dp[][] = new int[goodsSize][weight + 1];
        dp[0][0] = 0;
        //å¦‚æœèƒŒåŒ…å®¹é‡æ˜¯0,ä»·å€¼ä¹Ÿæ˜¯0
        for (int i = 1; i < dp.length; i++) {
            dp[i][0] = 0;
        }
        //åˆå§‹åŒ–ä»·å€¼
        for (int i = 1; i < dp[0].length; i++) {
            //å¦‚æœç¬¬ä¸€ä¸ªå•†å“çš„é‡é‡å°äºèƒŒåŒ…é‡é‡çš„è¯ï¼Œé‚£ä¹ˆå°±é»˜è®¤å˜ä¸º0
            if (i < nums[0]) {
                dp[0][i] = 0;
            } else {
                //å¦åˆ™å°±æ˜¯è‡ªå·±çš„é‡é‡
                dp[0][i] = nums[0];
            }
        }

        for (int i = 1; i < dp.length; i++) {
            for (int j = 1; j < dp[0].length; j++) {
                //å½“å‰çš„èƒŒåŒ…å®¹é‡<è‡ªèº«é‡é‡ï¼Œå°±ç”¨ä¸Šä¸€è¡Œçš„
                if (j < nums[i]) {

                  //å¦åˆ™è¯´æ˜èƒŒåŒ…å®¹é‡å¤Ÿ

                }
            }
        }

        return dp[goodsSize - 1][weight] == weight ? true : false;
    }
