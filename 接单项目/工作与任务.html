          title: '收支金额',
          type: 'number',
          search: { show: true },
          column: {
            show: true, width: 160,
            // 显示在列中的格式化器，确保显示两位小数
            formatter: (row, column, cellValue) => {
              if (row.value !== undefined && row.value !== null && !isNaN(row.value)) {
                return parseFloat(row.value).toFixed(2);
              }
              return row.value;
            }
          },
          form: {
            show: true,
            rules: [
              { required: true, message: "请输入收支金额" },
              {
                validator: (rule, value) => {
                  return new Promise((resolve, reject) => {
                    if (value !== null && value !== undefined && !isNaN(value)) {
                      const valueString = value.toString();
                      if (valueString.includes('.') && valueString.split('.')[1].length > 2) {
                        reject(new Error('小数点后不能超过两位'));
                      } else {
                        resolve();
                      }
                    } else {
                      resolve();
                    }
                  });
                },
                trigger: 'blur'
              }
            ],
            component: {
              props: {
                  //一次增加只加0.01
                step: 0.01
              }
            }
          }
        },
