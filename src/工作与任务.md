</template>

		:isEditMode="isEditMode"： 父组件传值给子组件

       </Modal>

    //省略其它
    
    //导入组件并注册组件
    import AddForm from './dealuser/addForm.vue';
    const components = {
        AddForm,
    };
    
    //声明对话框默认不显示，可通过其它处理事件设置showAddForm.value=true更改值为显示
    const showAddForm = ref(false);
    
    //声明子组件的引用
    const addFormRef = ref();
	
    //是否是编辑状态
	const isEditMode = ref(false);
    

    //对话框点击确认后的处理事件
    const handleAddSelectedRow = () => {
        
        //这里可以调用子组件中暴露出来的方法进行业务逻辑
        addFormRef.value.xxx方法()
        
        //关闭对话框显示
		showAddForm.value = false;
  };
    
    
    //编辑按钮触发的时候
    const editRow = (record) => {
        //父组件设置值传值给子组件
      isEditMode.value = true;
      showAddForm.value = true;
      nextTick(() => {
          //其它逻辑
        });
    };


  //更改显示的值
  showAddForm.value = false;
}
</script>
