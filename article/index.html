public int[] sortedSquares(int[] nums) {
    int result[] = new int[nums.length];
    //é¦–å°¾æŒ‡é’ˆ
    int startIndex = 0;
    int endIndex = nums.length - 1;
    int size = nums.length - 1;

    while (size >= 0) {
        //å› ä¸ºnumsæ˜¯æœ‰åºæ•°ç»„,æ‰€ä»¥å·¦å³ä¸¤è¾¹çš„å¹³æ–¹ä¸€å®šæ¯”ä¸­é—´å¤§,ä»ä¸¤è¾¹å¾€ä¸­é—´éå†ï¼ŒæŠŠæœ€å¤§çš„æ”¾åœ¨æ–°æ•°ç»„å³è¾¹
